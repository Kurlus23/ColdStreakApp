import React, { useState, useEffect } from "react";
import { Card, CardContent } from "@/components/ui/card";
import { Button } from "@/components/ui/button";

/*
OpenPlunge MVP+++ 
Features:
- Stopwatch timer (auto log on stop)
- Countdown timer with alarm (auto log on completion)
- Multiple screens (Timer / Countdown / History)
- Default screen memory
- Plunge streak tracking
- Weekly exposure tracking
- Improved brown fat scoring
- Timer input via dropdowns (minutes & seconds)
- History logs deletion
- Bluetooth thermometer integration
*/

export default function ColdPlungeApp() {

  const todayString = new Date().toLocaleDateString();

  const [screen, setScreen] = useState(localStorage.getItem("defaultScreen") || "timer");

  const setDefaultScreen = (name) => {
    setScreen(name);
    localStorage.setItem("defaultScreen", name);
  };

  const [seconds, setSeconds] = useState(0);
  const [running, setRunning] = useState(false);

  const [countdown, setCountdown] = useState(0);
  const [countdownRunning, setCountdownRunning] = useState(false);

  const [minutesInput, setMinutesInput] = useState(3);
  const [secondsInput, setSecondsInput] = useState(0);

  const [temp, setTemp] = useState(50);

  const [log, setLog] = useState(JSON.parse(localStorage.getItem("plungeLog") || "[]"));

  /* Stopwatch timer */
  useEffect(() => {
    let interval;
    if (running) {
      interval = setInterval(() => {
        setSeconds((s) => s + 1);
      }, 1000);
    }
    return () => clearInterval(interval);
  }, [running]);

  /* Auto log stopwatch when paused */
  const pauseTimer = () => {
    if (running && seconds > 0) {
      logSession(seconds);
    }
    setRunning(false);
  };

  /* Countdown timer */
  useEffect(() => {
    let interval;
    if (countdownRunning && countdown > 0) {
      interval = setInterval(() => {
        setCountdown((c) => c - 1);
      }, 1000);
    }
    if (countdownRunning && countdown === 0) {
      setCountdownRunning(false);
      logSession(minutesInput * 60 + secondsInput);
      const audio = new Audio("https://actions.google.com/sounds/v1/alarms/alarm_clock.ogg");
      audio.play();
      alert("Plunge finished!");
    }
    return () => clearInterval(interval);
  }, [countdownRunning, countdown]);

  /* Improved plunge score */
  const plungeScore = (time, temperature) => {
    const minutes = time / 60;
    let coldFactor = 1;
    if (temperature <= 55) coldFactor = 1.2;
    if (temperature <= 50) coldFactor = 1.5;
    if (temperature <= 45) coldFactor = 1.9;
    if (temperature <= 40) coldFactor = 2.3;
    return Number((minutes * coldFactor).toFixed(2));
  };

  /* Log a plunge */
  const logSession = (duration) => {
    const entry = {
      date: todayString,
      duration: duration,
      temperature: temp,
      score: plungeScore(duration, temp)
    };
    const updated = [entry, ...log];
    setLog(updated);
    localStorage.setItem("plungeLog", JSON.stringify(updated));
    setSeconds(0);
    setCountdown(0);
  };

  const formatTime = (s) => {
    const m = Math.floor(s / 60);
    const sec = s % 60;
    return `${m}:${sec.toString().padStart(2, "0")}`;
  };

  const startCountdown = () => {
    const total = minutesInput * 60 + secondsInput;
    setCountdown(total);
    setCountdownRunning(true);
  };

  /* Stats */
  const todayLog = log.filter((l) => l.date === todayString);
  const todayTotal = todayLog.reduce((sum, l) => sum + l.duration, 0);
  const todayScore = todayLog.reduce((sum, l) => sum + Number(l.score), 0);
  const last7Days = log.slice(0,50).filter((entry) => (new Date() - new Date(entry.date))/(1000*60*60*24) <= 7);
  const weeklyMinutes = last7Days.reduce((sum, l) => sum + l.duration,0)/60;
  const weeklyGoal = 11;

  /* Streak */
  const getStreak = () => {
    let streak = 0;
    const dates = [...new Set(log.map((l)=>l.date))];
    const sorted = dates.map(d=>new Date(d)).sort((a,b)=>b-a);
    let current = new Date();
    for(let d of sorted){
      const diff = Math.floor((current-d)/(1000*60*60*24));
      if(diff <= 1){
        streak++;
        current=d;
      }else break;
    }
    return streak;
  };
  const streak = getStreak();

  /* Delete a log entry */
  const deleteLog = (index) => {
    const updated = [...log];
    updated.splice(index,1);
    setLog(updated);
    localStorage.setItem("plungeLog", JSON.stringify(updated));
  };

  /* Bluetooth thermometer */
  const connectThermometer = async () => {
    try {
      const device = await navigator.bluetooth.requestDevice({
        filters: [{ services: ['health_thermometer'] }]
      });
      const server = await device.gatt.connect();
      const service = await server.getPrimaryService('health_thermometer');
      const characteristic = await service.getCharacteristic('temperature_measurement');
      characteristic.startNotifications();
      characteristic.addEventListener('characteristicvaluechanged', (event) => {
        const value = event.target.value;
        let tempC = value.getUint8(1); // example parse
        let tempF = (tempC * 9/5) + 32;
        setTemp(tempF);
      });
      alert('Thermometer connected!');
    } catch (err) {
      console.error('Bluetooth error:', err);
      alert('Could not connect to thermometer.');
    }
  };

  return (
    <div className="p-6 grid gap-6 max-w-xl mx-auto">

      {/* Navigation */}
      <Card>
        <CardContent className="p-4 flex justify-around">
          <Button onClick={() => setDefaultScreen("timer")}>Timer</Button>
          <Button onClick={() => setDefaultScreen("countdown")}>Countdown</Button>
          <Button onClick={() => setDefaultScreen("history")}>History</Button>
        </CardContent>
      </Card>

      {/* Stats */}
      <Card>
        <CardContent className="p-4 text-center text-sm">
          <div>🔥 Streak: {streak} days</div>
          <div>Today: {(todayTotal/60).toFixed(1)} min</div>
          <div>Cold Score Today: {todayScore.toFixed(2)}</div>
          <div>Weekly Exposure: {weeklyMinutes.toFixed(1)} / {weeklyGoal} min</div>
        </CardContent>
      </Card>

      {/* Timer Screen */}
      {screen==="timer" && (
        <Card>
          <CardContent className="p-6 text-center">
            <h2 className="text-xl font-bold mb-4">Cold Plunge Timer</h2>
            <div className="text-5xl font-mono mb-4">{formatTime(seconds)}</div>
            <div className="flex gap-3 justify-center mb-4">
              <Button onClick={()=>setRunning(true)}>Start</Button>
              <Button onClick={pauseTimer}>Pause & Auto Log</Button>
              <Button onClick={()=>{setSeconds(0); setRunning(false);}}>Reset</Button>
            </div>
            <div className="mb-4">
              <label className="text-sm">Water Temp (°F)</label>
              <input type="number" value={temp} onChange={e=>setTemp(Number(e.target.value))} className="border rounded p-2 w-24 ml-2"/>
            </div>
            <Button onClick={connectThermometer}>Connect Thermometer</Button>
          </CardContent>
        </Card>
      )}

      {/* Countdown Screen */}
      {screen==="countdown" && (
        <Card>
          <CardContent className="p-6 text-center">
            <h2 className="text-xl font-bold mb-4">Countdown Timer</h2>
            <div className="text-5xl font-mono mb-4">{formatTime(countdown)}</div>
            <div className="flex justify-center gap-2 mb-4">
              <select value={minutesInput} onChange={e=>setMinutesInput(Number(e.target.value))} className="border p-2">
                {Array.from({length:61},(_,i)=><option key={i} value={i}>{i} min</option>)}
              </select>
              <select value={secondsInput} onChange={e=>setSecondsInput(Number(e.target.value))} className="border p-2">
                {Array.from({length:60},(_,i)=><option key={i} value={i}>{i} sec</option>)}
              </select>
            </div>
            <div className="flex justify-center gap-3">
              <Button onClick={startCountdown}>Start</Button>
              <Button onClick={()=>{setCountdownRunning(false); setCountdown(0);}}>Reset</Button>
            </div>
          </CardContent>
        </Card>
      )}

      {/* History Screen */}
      {screen==="history" && (
        <Card>
          <CardContent className="p-6">
            <h2 className="text-xl font-bold mb-4">Plunge History</h2>
            {log.length===0 && <div>No plunges logged yet.</div>}
            {log.map((l,i)=>(
              <div key={i} className="border-b py-2 text-sm flex justify-between items-center">
                <div>
                  {l.date} — {(l.duration/60).toFixed(1)} min @ {l.temperature}°F | Score: {l.score}
                </div>
                <Button onClick={()=>deleteLog(i)} className="text-red-500">Delete</Button>
              </div>
            ))}
          </CardContent>
        </Card>
      )}

    </div>
  );
}
